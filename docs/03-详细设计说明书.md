### 1.3 protocol/ 模块 trait 定义

```rust
// === protocol/mod.rs ===

use async_trait::async_trait;
use crate::error::AppResult;

/// 统一消息协议处理器 trait
/// 只负责协议解析和数据构建，不涉及 CDP/IPC
/// CDP 连接管理由 ProtocolCoordinator 负责
#[async_trait]
pub trait MessageHandler: Send + Sync {
    /// 平台标识
    fn platform(&self) -> Platform;

    /// 初始化协议解析器（加载所需资源，不依赖 CDP）
    async fn init(&self) -> AppResult<()>;

    /// 解析原始数据（来自 CDP 的 WebSocket 帧或 HTTP 响应）
    /// 纯函数式，无副作用
    fn parse_raw_data(&self, data: &[u8], data_type: DataType) -> AppResult<Option<Message>>;

    /// 获取消息列表（翻页）
    async fn fetch_messages(
        &self,
        session_id: &str,
        cursor: Option<String>,
        limit: u32,
    ) -> AppResult<Vec<Message>>;

    /// 构建发送消息的请求数据
    /// 返回需要通过 CDP 发送的原始字节
    fn build_send_request(
        &self,
        session_id: &str,
        content: &MessageContent,
    ) -> AppResult<Vec<u8>>;

    /// 获取联系人列表
    async fn fetch_contacts(&self) -> AppResult<Vec<Contact>>;

    /// 获取会话列表
    async fn fetch_sessions(&self, cursor: Option<String>) -> AppResult<Vec<Session>>;
}

// ===== CDP 事件协调器 =====

/// CDP 事件协调器
/// 管理 CDP 连接和协议解析器之间的数据流转
pub struct ProtocolCoordinator {
    handler: Box<dyn MessageHandler>,
    cdp_client: CdpClient,
    message_sender: mpsc::UnboundedSender<Message>,
}

impl ProtocolCoordinator {
    /// 创建协调器
    pub fn new(
        handler: Box<dyn MessageHandler>,
        cdp_client: CdpClient,
    ) -> Self;

    /// 启动 CDP 事件监听
    pub async fn start(&mut self) -> AppResult<()>;

    /// 停止监听
    pub async fn stop(&mut self) -> AppResult<()>;

    /// 处理 CDP 网络帧（收到事件后调用 handler.parse_raw_data）
    pub async fn on_network_frame(&self, frame: CdpNetworkFrame) -> AppResult<Option<Message>>;
}

/// CDP 网络帧数据类型
#[derive(Debug, Clone)]
pub enum CdpNetworkFrame {
    /// WebSocket 帧（WhatsApp WebSocket, Telegram MTProto）
    WebSocket {
        request_id: String,
        payload: Vec<u8>,
        is_binary: bool,
    },
    /// 网络响应（GraphQL, REST API）
    HttpResponse {
        url: String,
        method: String,
        request_body: Option<Vec<u8>>,
        response_body: Vec<u8>,
        status_code: u16,
    },
}

/// 数据类型（传递给 MessageHandler.parse_raw_data）
#[derive(Debug, Clone, Copy, PartialEq, Eq)]
pub enum DataType {
    WebSocketFrame,
    HttpResponse,
}

/// 协议解析器注册中心
pub struct ProtocolRegistry {
    handlers: HashMap<Platform, Box<dyn MessageHandler>>,
}

impl ProtocolRegistry {
    pub fn new() -> Self {
        Self { handlers: HashMap::new() }
    }

    pub fn register(&mut self, handler: Box<dyn MessageHandler>) {
        let platform = handler.platform();
        self.handlers.insert(platform, handler);
    }

    pub fn get(&self, platform: &Platform) -> Option<&Box<dyn MessageHandler>> {
        self.handlers.get(platform)
    }
}
```
